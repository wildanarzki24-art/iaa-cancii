<script>
const pesanList = [
"Semangatt ya aigaa novariaa 💪💞",
"idann bangga sama sayangg 🥰",
"sayangg hebat ❤️",
"Sedikit lagi pasti bisa 🔥",
"idann selalu dukung sayangg 💞"
];

function gantiPesan() {
    const random = Math.floor(Math.random() * pesanList.length);
    document.getElementById("pesan").innerText = pesanList[random];
}

/* efek hati */
setInterval(() => {
    const heart = document.createElement("div");
    heart.className = "heart";
    heart.innerText = "💝";
    heart.style.left = Math.random() * 100 + "vw";
    document.body.appendChild(heart);

    setTimeout(() => heart.remove(), 5000);
}, 500);
